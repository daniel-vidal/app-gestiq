CREATE TABLE mod_scraper.regioes (
    id SERIAL PRIMARY KEY,
    nome VARCHAR(150) NOT NULL,
    cidade VARCHAR(120) NOT NULL,
    estado CHAR(2) NOT NULL,
    tipo_regiao VARCHAR(50), -- bairro, litoral, ilha, orla, centro etc
    ativa BOOLEAN NOT NULL DEFAULT TRUE,
    criada_em TIMESTAMP NOT NULL DEFAULT NOW(),
    atualizada_em TIMESTAMP NOT NULL DEFAULT NOW()
);

CREATE UNIQUE INDEX ux_mod_scraper_regioes_nome_cidade_estado
    ON mod_scraper.regioes (nome, cidade, estado);

---------------------------------------------------

CREATE TABLE mod_scraper.hoteis_monitorados (
    id SERIAL PRIMARY KEY,
    nome VARCHAR(200) NOT NULL,
    regiao_id INTEGER NOT NULL REFERENCES mod_scraper.regioes(id),
    categoria_estrelas SMALLINT, -- 1 a 5
    perfil_hotel VARCHAR(80), -- boutique, luxo, resort, pousada, economico etc
    url_booking TEXT NOT NULL,
    ativo BOOLEAN NOT NULL DEFAULT TRUE,
    prioridade_monitoramento SMALLINT NOT NULL DEFAULT 5, -- 1 maior prioridade
    hotel_base BOOLEAN NOT NULL DEFAULT FALSE,
    observacoes TEXT,
    criada_em TIMESTAMP NOT NULL DEFAULT NOW(),
    atualizada_em TIMESTAMP NOT NULL DEFAULT NOW()
);

CREATE UNIQUE INDEX ux_mod_scraper_hoteis_url_booking
    ON mod_scraper.hoteis_monitorados (url_booking);

CREATE INDEX ix_mod_scraper_hoteis_regiao
    ON mod_scraper.hoteis_monitorados (regiao_id);

CREATE INDEX ix_mod_scraper_hoteis_ativo
    ON mod_scraper.hoteis_monitorados (ativo);

CREATE TABLE IF NOT EXISTS mod_scraper.rotinas_agendadas (
id BIGSERIAL PRIMARY KEY,
nome VARCHAR(150) NOT NULL,
tipo_rotina VARCHAR(60) NOT NULL,
ativo BOOLEAN NOT NULL DEFAULT TRUE,
script_gerador VARCHAR(255) NOT NULL,
parametros_json JSONB NOT NULL DEFAULT '{}'::jsonb,
frequencia_tipo VARCHAR(30) NOT NULL,
frequencia_valor INTEGER NOT NULL DEFAULT 1,
hora_inicio TIME NULL,
hora_fim TIME NULL,
dias_semana SMALLINT[] NULL,
prioridade INTEGER NOT NULL DEFAULT 5,
ultima_execucao_em TIMESTAMP NULL,
proxima_execucao_em TIMESTAMP NULL,
total_execucoes INTEGER NOT NULL DEFAULT 0,
total_dias_execucao INTEGER NOT NULL DEFAULT 0,
criado_em TIMESTAMP NOT NULL DEFAULT NOW(),
atualizada_em TIMESTAMP NOT NULL DEFAULT NOW(),
CONSTRAINT ck_rotinas_agendadas_frequencia_tipo
CHECK (frequencia_tipo IN ('minutos', 'horas', 'diaria', 'semanal', 'mensal')),
CONSTRAINT ck_rotinas_agendadas_frequencia_valor
CHECK (frequencia_valor > 0),
CONSTRAINT ck_rotinas_agendadas_prioridade
CHECK (prioridade >= 0)
);


CREATE INDEX IF NOT EXISTS ix_rotinas_agendadas_ativo_proxima
    ON mod_scraper.rotinas_agendadas (ativo, proxima_execucao_em);

CREATE INDEX IF NOT EXISTS ix_rotinas_agendadas_tipo
    ON mod_scraper.rotinas_agendadas (tipo_rotina);


---------------------------------------------------


CREATE TABLE IF NOT EXISTS mod_scraper.rotinas_agendadas_execucoes (
id BIGSERIAL PRIMARY KEY,
rotina_id BIGINT NOT NULL REFERENCES mod_scraper.rotinas_agendadas(id) ON DELETE CASCADE,
executada_em TIMESTAMP NOT NULL DEFAULT NOW(),
status VARCHAR(30) NOT NULL,
mensagem TEXT NULL,
tarefas_geradas INTEGER NOT NULL DEFAULT 0,
duracao_ms BIGINT NULL,
erro TEXT NULL,
criado_em TIMESTAMP NOT NULL DEFAULT NOW(),
CONSTRAINT ck_rotinas_agendadas_execucoes_status
CHECK (status IN ('sucesso', 'erro', 'ignorada'))
);


CREATE INDEX IF NOT EXISTS ix_rotinas_agendadas_execucoes_rotina
    ON mod_scraper.rotinas_agendadas_execucoes (rotina_id, executada_em DESC);


---------------------------------------------------


CREATE OR REPLACE FUNCTION mod_scraper.fn_set_atualizada_em()
RETURNS TRIGGER AS $$
BEGIN
    NEW.atualizada_em = NOW();
    RETURN NEW;
END;
$$ LANGUAGE plpgsql;

DROP TRIGGER IF EXISTS trg_rotinas_agendadas_set_atualizada_em
ON mod_scraper.rotinas_agendadas;

CREATE TRIGGER trg_rotinas_agendadas_set_atualizada_em
BEFORE UPDATE ON mod_scraper.rotinas_agendadas
FOR EACH ROW
EXECUTE FUNCTION mod_scraper.fn_set_atualizada_em();
